<template>
  <label
    class="field-label"
    :class="cssClass"
    :for="targetName"
    @click="onClickLabel()"
  >
    <!-- @slot Label of the field -->
    <slot />
    <mks-icon :type="icon" v-if="icon" class="field-label__icon" />
  </label>
</template>
 
<script>
/**
 * The field label is used in the Field component
 * @version 1.0.0
 */
export default {
  name: "MksFieldLabel",
  props: {
    /**
     * The name of the targeted field
     */
    targetName: {
      type: String,
      required: false
    },
    /**
     * The css class from the field component
     */
    cssClass: {
      type: [String, Array, Object],
      required: false
    },
    /**
     * The icon to use (any icon available in the icon component)
     */
    icon: {
      type: String,
      required: false,
      default: null
    }
  },
  computed: {},
  methods: {
    onClickLabel: function() {
      this.$emit("click");
    }
  }
};
</script>
